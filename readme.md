# Installation Tailwind v 3.3.3
npm install -D tailwindcss@3.3.3
npx tailwindcss init

# Pour appliquer le style tailwind s'il ne fonctionne pas 
npx tailwindcss -i ./css/tailwind.css -o ./css/tailwind.output.css --watch

CTRL + S 2 fois dans l'html pour actualiser
